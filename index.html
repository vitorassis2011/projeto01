import sqlite3 from datetime import datetime # Conexão com o banco de dados SQLite def conectar_db(): conn = sqlite3.connect('monitoramento.db') return conn # Criação das tabelas no banco de dados def criar_tabelas(): conn = conectar_db() cursor = conn.cursor() cursor.execute(''' CREATE TABLE IF NOT EXISTS empresas ( id INTEGER PRIMARY KEY, nome TEXT NOT NULL, cnpj TEXT NOT NULL UNIQUE, data_criacao TEXT, status TEXT ); ''') cursor.execute(''' CREATE TABLE IF NOT EXISTS monitoramento ( id INTEGER PRIMARY KEY, empresa_id INTEGER, data_monitora TEXT, status_monitoramento TEXT, FOREIGN KEY (empresa_id) REFERENCES empresas(id) ); ''') conn.commit() conn.close() # Função para cadastrar uma nova empresa def cadastrar_empresa(nome, cnpj): conn = conectar_db() cursor = conn.cursor() data_criacao = datetime.now().strftime('%Y-%m-%d %H:%M:%S') status = 'Ativa' # Status inicial da empresa cursor.execute(''' INSERT INTO empresas (nome, cnpj, data_criacao, status) VALUES (?, ?, ?, ?) ''', (nome, cnpj, data_criacao, status)) conn.commit() conn.close() # Função para monitorar o status de uma empresa def monitorar_empresa(empresa_id, status_monitoramento): conn = conectar_db() cursor = conn.cursor() data_monitoramento = datetime.now().strftime('%Y-%m-%d %H:%M:%S') # Registra o monitoramento cursor.execute(''' INSERT INTO monitoramento (empresa_id, data_monitora, status_monitoramento) VALUES (?, ?, ?) ''', (empresa_id, data_monitoramento, status_monitoramento)) # Atualiza o status da empresa no banco de dados cursor.execute(''' UPDATE empresas SET status = ? WHERE id = ? ''', (status_monitoramento, empresa_id)) conn.commit() conn.close() # Função para gerar relatório de monitoramento def gerar_relatorio(): conn = conectar_db() cursor = conn.cursor() cursor.execute(''' SELECT empresas.nome, empresas.status, monitoramento.data_monitora, monitoramento.status_monitoramento FROM empresas LEFT JOIN monitoramento ON empresas.id = monitoramento.empresa_id ORDER BY monitoramento.data_monitora DESC; ''') relatorio = cursor.fetchall() print("Relatório de Monitoramento de Empresas:") print("Nome da Empresa | Status Atual | Data Monitoramento | Status do Monitoramento") print("-" * 80) for row in relatorio: print(f"{row[0]} | {row[1]} | {row[2]} | {row[3]}") conn.close() # Função principal def main(): criar_tabelas() # Cadastrando algumas empresas para testes cadastrar_empresa('Tech Solutions Ltda', '12.345.678/0001-99') cadastrar_empresa('InovaCorp', '98.765.432/0001-00') # Monitorando o status das empresas monitorar_empresa(1, 'Em Manutenção') # Empresa 1 em manutenção monitorar_empresa(2, 'Ativa') # Empresa 2 está ativa # Gerando relatório gerar_relatorio() if __name__ == '__main__': main()
