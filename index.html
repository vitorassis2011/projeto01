import pygame import sys # --- CONFIGURAÇÕES GLOBAIS --- SCREEN_WIDTH = 1200 SCREEN_HEIGHT = 700 FPS = 60 # Cores WHITE = (255, 255, 255) BLUE = (100, 149, 237) # Cor do céu RED = (255, 0, 0) BROWN = (139, 69, 19) GREEN = (34, 139, 34) # --- CLASSES PRINCIPAIS --- class Player(pygame.sprite.Sprite): def __init__(self, x, y): super().__init__() self.image = pygame.Surface((40, 60)) self.image.fill(RED) self.rect = self.image.get_rect() self.rect.x = x self.rect.y = y # Vetores de movimento self.change_x = 0 self.change_y = 0 self.speed = 6 self.jump_force = -15 self.gravity = 0.8 def update(self, platforms): # Aplicar Gravidade self.calc_grav() # Movimento Horizontal self.rect.x += self.change_x block_hit_list = pygame.sprite.spritecollide(self, platforms, False) for block in block_hit_list: if self.change_x > 0: self.rect.right = block.rect.left elif self.change_x < 0: self.rect.left = block.rect.right # Movimento Vertical self.rect.y += self.change_y block_hit_list = pygame.sprite.spritecollide(self, platforms, False) for block in block_hit_list: if self.change_y > 0: self.rect.bottom = block.rect.top self.change_y = 0 elif self.change_y < 0: self.rect.top = block.rect.bottom self.change_y = 0 def calc_grav(self): if self.change_y == 0: self.change_y = 1 else: self.change_y += self.gravity def jump(self, platforms): # Verifica se o jogador está no chão para pular self.rect.y += 2 hit_list = pygame.sprite.spritecollide(self, platforms, False) self.rect.y -= 2 if len(hit_list) > 0: self.change_y = self.jump_force def go_left(self): self.change_x = -self.speed def go_right(self): self.change_x = self.speed def stop(self): self.change_x = 0 class Platform(pygame.sprite.Sprite): def __init__(self, x, y, width, height): super().__init__() self.image = pygame.Surface((width, height)) self.image.fill(GREEN) self.rect = self.image.get_rect() self.rect.x = x self.rect.y = y class Level: def __init__(self, player): self.platform_list = pygame.sprite.Group() self.player = player self.world_shift = 0 # Para o efeito de câmera def update(self): self.platform_list.update() def draw(self, screen): screen.fill(BLUE) self.platform_list.draw(screen) def shift_world(self, shift_x): self.world_shift += shift_x for platform in self.platform_list: platform.rect.x += shift_x # --- LOOP PRINCIPAL --- def main(): pygame.init() screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Super Mario Clone - Engine") clock = pygame.time.Clock() player = Player(100, SCREEN_HEIGHT - 150) level = Level(player) # Criando o cenário (Chão e plataformas) level_data = [ [0, 650, 2000, 50], [500, 500, 200, 40], [800, 400, 200, 40], [1100, 300, 200, 40], [1400, 500, 400, 40], ] for plat in level_data: block = Platform(plat[0], plat[1], plat[2], plat[3]) level.platform_list.add(block) active_sprite_list = pygame.sprite.Group() active_sprite_list.add(player) running = True while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.KEYDOWN: if event.key == pygame.K_LEFT: player.go_left() if event.key == pygame.K_RIGHT: player.go_right() if event.key == pygame.K_SPACE: player.jump(level.platform_list) if event.type == pygame.KEYUP: if event.key == pygame.K_LEFT and player.change_x < 0: player.stop() if event.key == pygame.K_RIGHT and player.change_x > 0: player.stop() # Atualização player.update(level.platform_list) # Lógica da Câmera (Scrolling) if player.rect.right >= 500: diff = player.rect.right - 500 player.rect.right = 500 level.shift_world(-diff) if player.rect.left <= 120: diff = 120 - player.rect.left player.rect.left = 12
