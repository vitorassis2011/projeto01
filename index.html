Login dos jogadores nome1, senha1 = login() nome2, senha2 = login() # Loop principal while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() keys = pygame.key.get_pressed() # Garante que vai para a direita # Controles Jogador 2 (Setas + K para chutar) if keys[pygame.K_LEFT] and player2.left > 0: player2.x -= speed if keys[pygame.K_RIGHT] and player2.right < WIDTH: player2.x += speed if keys[pygame.K_UP] and player2.bottom >= HEIGHT: p2_y_vel = -jump_height if keys[pygame.K_k] and player2.colliderect(ball): ball_speed[0] = -abs(ball_speed[0]) * 1.5 #
